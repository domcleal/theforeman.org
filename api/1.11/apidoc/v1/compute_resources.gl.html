POST /api/compute_resources
Create a compute resource.

Examples

POST /api/compute_resources
{
  "compute_resource": {
    "name": "special_compute",
    "provider": "EC2",
    "region": "eu-west-1",
    "user": "user@example.com",
    "password": "secret"
  }
}
201
{
  "compute_resource": {
    "id": 980190963,
    "name": "special_compute",
    "description": null,
    "url": "eu-west-1",
    "created_at": "2016-03-10 15:51:25 UTC",
    "updated_at": "2016-03-10 15:51:25 UTC",
    "provider": "EC2",
    "user": "user@example.com",
    "region": "eu-west-1"
  }
}
